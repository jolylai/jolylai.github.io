src/app/app.module.ts

  • MockOption
    • data: any = ‘’ mock数据规矩
    • delay: number = 300 请求延迟毫秒数
    • force: boolean = false 是否强制所有请求都Mock，true 表示当请求的URL不存在时直接返回 404 错误，false 表示未命中时发送真实HTTP请求
    • log: boolean = true 是否打印 Mock 请求信息，弥补浏览器无Network信息
